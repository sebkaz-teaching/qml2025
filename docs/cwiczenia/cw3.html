    """Generuje histogram, wy≈õwietlajƒÖc wszystkie 4 stany (00, 01, 10, 11)."""
    full_counts = {state: counts.get(state, 0) for state in ALL_STATES}
    probabilities = [full_counts[state] / SHOTS for state in ALL_STATES]
    
    plt.figure(figsize=(7, 5))

    plt.title(title + f" ({SHOTS} shots)")

    plt.grid(axis='y', alpha=0.5)

    for i, prob in enumerate(probabilities):
        plt.text(i, prob + 0.02, f'{prob:.3f}', ha='center', fontsize=10)

Kopiowanie Kubitu

Klasyczne komputery bardzo czƒôsto wykorzystujƒÖc operacjƒô kopiowania.

Zobaczmy jak taka operacja wyglƒÖda dla kubit√≥w.

Rozwazmy obwod z operatorem C, kt√≥ry w dzia≈Çaniu na dwa kubity kopiuje warto≈õƒá pierwszego kubitu na wynik drugiego. Drugi kubit mozna na poczƒÖtku ustawiƒá w dowolnym stanie.

Chcemy skopiowaƒá stan \(\ket{\psi_0} = a\ket{0} + b\ket{1}\)

Stan poczƒÖtkowy uk≈Çadu: \(\ket{\psi_0} \otimes \ket{0}\)

Chcemy przekszta≈Çciƒá na \(\ket{\psi_0} \otimes \ket{\psi_0}\) czyli

\[ C \left(\ket{\psi_0} \otimes \ket{0}\right) = \ket{\psi_0} \otimes \ket{\psi_0} \]

Lewa strona

\[ C \left(\ket{\psi_0} \otimes \ket{0}\right) = C\left( (a\ket{0} + b\ket{1} ) \otimes \ket{0} \right) \] \[ C\left( a\ket{0} \otimes \ket{0} + b\ket{1}\otimes \ket{0} \right) = a C \left(\ket{0} \otimes \ket{0}\right) + b C \left( \ket{1}\otimes \ket{0}\right) \] \[ a \ket{00} + b \ket{11} \]

Prawa strona \[ \ket{\psi_0} \otimes \ket{\psi_0} = a^2 \ket{00} + ab\ket{01} + ab\ket{10} + b^2\ket{11} \]

Napisz operator 1+1 na uk≈Çadzie 4 kubit√≥w

\[ 0+0 = 00 \] \[ 0+1 = 01 \] \[ 1+0 = 01 \] \[ 1+1 = 10 \]

zauwaz, ze mamy dwa typy rozwiƒÖza≈Ñ:

  • dwa bity wejsciowe sƒÖ takie same (00, 11) i dajƒÖ na prawym bicie odpowiedzi 0.
  • dwa bity wejsciowe sƒÖ r√≥zne (10,01) i dajƒÖ na prawym bicie odpowiedzi 1.

Bramka (i operator) Z, w bazie obliczeniowej dany jest macierzƒÖ: \[ \textbf{Z} = \begin{bmatrix} 1 \,\,\,\,\,\,\,\, 0 \\ 0 \,\, -1 \end{bmatrix} \]

Operator ten mierzy r√≥≈ºnicƒô pomiƒôdzy prawdopodobie≈Ñstwem, ≈ºe kubit jest w stanie \(\ket{0}\) a prawdopodobie≈Ñstwem, ≈ºe jest w stanie \(\ket{1}\)

W og√≥lno≈õci warto≈õƒá oczekiwana (warto≈õƒá ≈õrednia wyniku pomiaru w bazie operatora Z) dana jest wzorem: \[ \textbf{<Z>} = \bra{\psi} \textbf{Z} \ket{\psi} \]

Niech \[ \ket{\psi} = \alpha\ket{0} + \beta\ket{1} \] wtedy \[ \bra{\psi} = \alpha^*\bra{0} + \beta^*\bra{1} \]

Mo≈ºemy obliczyƒá: \[ \bra{\psi} \textbf{Z} \ket{\psi} = (\alpha^*\bra{0} + \beta^*\bra{1} ) \,\,\, Z \,\,\,(\alpha\ket{0} + \beta\ket{1}) = |\alpha|^2 - |\beta|^2 \] Czyli dla kubitu w stanie \(\ket{0}\) \[ \textbf{<Z>} = 1 \] Dla kubitu w stanie \(\ket{1}\) \[ \textbf{<Z>} = -1 \] Dla kubitu w superpozycji \(\ket{0} +\ket{1}\) \[ \textbf{<Z>} = 0 \]

üßê Wyja≈õnienie Dzia≈Çania Funkcji Kosztu w VQA (Mini-Eksperyment)

W tym prostym eksperymencie z Wariacyjnym Algorytmem Kwantowym (VQA), optymalizator (Gradient Descent) minimalizuje funkcjƒô kosztu, choƒá nie zosta≈Ça ona jawnie zdefiniowana jako np. Mean Squared Error (MSE). Dzieje siƒô tak, poniewa≈º w bibliotekach takich jak PennyLane, optymalizator domy≈õlnie przyjmuje, ≈ºe to, co zwraca \(\mathbf{QNode}\) jest funkcjƒÖ celu, kt√≥rƒÖ nale≈ºy minimalizowaƒá.

1. Definicja Kosztu Przez Operator PauliZ

Tw√≥j obw√≥d qn(theta) jest zdefiniowany nastƒôpujƒÖco:

\[\mathbf{qn}(\theta) = \bra{\psi(\theta)} \mathbf{Z} \ket{\psi(\theta)} = \langle \mathbf{Z} \rangle\]

Poniewa≈º \(\mathbf{qn}(\theta)\) zwraca Warto≈õƒá OczekiwanƒÖ operatora PauliZ (\(\langle \mathbf{Z} \rangle \in [-1, 1]\)), optymalizator traktuje ten wynik jako funkcjƒô kosztu \(\mathbf{C}(\theta)\), kt√≥rƒÖ nale≈ºy sprowadziƒá do najni≈ºszej mo≈ºliwej warto≈õci, czyli \(-1\).

\[\text{Optymalizator dƒÖ≈ºy do minimalizacji: } \mathbf{C}(\theta) = \langle \mathbf{Z} \rangle\]

2. Cel Obwodu Kwantowego

W tym kontek≈õcie, obw√≥d VQA jest u≈ºywany do znalezienia stanu w≈Çasnego operatora \(\mathbf{Z}\) odpowiadajƒÖcego najni≈ºszej warto≈õci w≈Çasnej (\(\lambda = -1\)). Jest to analogiczne do algorytmu Variational Quantum Eigensolver (VQE), kt√≥ry s≈Çu≈ºy do znajdowania energii stanu podstawowego (najni≈ºszej warto≈õci w≈Çasnej Hamiltonianu).

  • Po≈ºƒÖdany Stan: Stan \(\ket{1}\), dla kt√≥rego \(\mathbf{Z}\ket{1} = -1\ket{1}\) (minimalna warto≈õƒá oczekiwana).
  • Architektura Obwodu:
    1. qml.X(wires=0): Ustawia kubit w stan \(\ket{1}\).
    2. qml.RY(theta, wires=0): Wprowadza parametr \(\theta\), kt√≥ry perturbuje stan \(\ket{1}\).
  • Dzia≈Çanie Optymalizatora: ZaczynajƒÖc z \(\theta\) bliskim zera (gdzie \(\langle \mathbf{Z} \rangle \approx -1\)), optymalizator bardzo szybko dƒÖ≈ºy do kƒÖta \(\theta\), kt√≥ry minimalizuje perturbacjƒô, czyli \(\theta \to 0\) lub \(\theta \to 2\pi\). W tych punktach stan wraca do \(\ket{1}\), a koszt osiƒÖga swoje minimum \(-1\).

üí° Podsumowanie

Eksperyment ten demonstruje podstawowƒÖ funkcjonalno≈õƒá wariacyjnego uczenia maszynowego w kwantowym kontek≈õcie, gdzie sam obw√≥d kwantowy mo≈ºe definiowaƒá funkcjƒô celu (Hamiltonian), a klasyczny optymalizator znajduje parametry \(\theta\), kt√≥re minimalizujƒÖ tƒô funkcjƒô. Jest to archetyp podej≈õcia hybrydowego (klasyczny optymalizator + kwantowe przetwarzanie).

import pennylane as qml
import pennylane.numpy as np
import matplotlib.pyplot as plt

# 1. Definicja UrzƒÖdzenia i Obwodu Kwantowego
dev = qml.device('default.qubit', wires=1)

@qml.qnode(dev)
def qn(theta):
    qml.X(wires=0)       # Ustawia stan na |1> (expval Z = -1)
    qml.RY(theta, wires=0) # Obr√≥t wok√≥≈Ç Y (zmienia expval Z)
    return qml.expval(qml.PauliZ(0)) # Funkcja celu: minimalizujemy expval Z

# 2. Inicjalizacja Parametr√≥w i Optymalizatora
# Zaczynamy z theta, gdzie koszt nie jest minimalny, aby zobaczyƒá proces optymalizacji.
# Np. theta = pi/2, gdzie RY(pi/2) na |1> daje 1/sqrt(2)(|1> - |0>), a expval Z = 0.
theta_init = np.array([np.pi-0.01], requires_grad=True) 
opt = qml.GradientDescentOptimizer(stepsize=0.1)

epochs = 100

# Listy do przechowywania historii treningu
cost_history = []
theta_history = []

# 3. Pƒôtla Treningowa
theta = theta_init # Ustawienie poczƒÖtkowej warto≈õci theta

print("--- Rozpoczƒôcie treningu ---")
for epoch in range(epochs):
    # Wykonaj krok optymalizatora i oblicz koszt
    theta, cost = opt.step_and_cost(qn, theta)
    
    # Zapisz historiƒô
    cost_history.append(cost)
    theta_history.append(theta[0]) # theta jest tablicƒÖ numpy, bierzemy tylko warto≈õƒá

    if epoch % 10 == 0 or epoch == epochs - 1:
        print(f"Epoka: {epoch:3d} | KƒÖt theta: {theta.item():.4f} | Koszt: {cost.item():.4f}")

print("\n--- Trening zako≈Ñczony ‚úÖ ---")
print(f"Ko≈Ñcowy kƒÖt theta: {theta.item():.4f}") 
print(f"Ko≈Ñcowy koszt: {cost.item():.4f}")

# 4. Wizualizacja Historii Treningu
fig, axes = plt.subplots(1, 2, figsize=(14, 5))

# Wykres kosztu
axes[0].plot(range(epochs), cost_history, color='blue', linewidth=2)
axes[0].set_title('Ewolucja Funkcji Kosztu')

import pennylane as qml 
import pennylane.numpy as np 
import matplotlib.pyplot as plt 
# 1. Definicja UrzƒÖdzenia (1 kubit) 
dev = qml.device('default.qubit', wires=1) 
# 2. Poprawiony Obw√≥d Kwantowy z ParametrycznƒÖ BramkƒÖ RX 
@qml.qnode(dev) 
def continuous_expval(theta): 
    """ Obw√≥d stosuje rotacjƒô RX z kƒÖtem theta. Zwraca warto≈õƒá oczekiwanƒÖ operatora PauliZ, kt√≥ra jest r√≥wna cos(theta). """ 
    # Zastosowanie bramki obrotu RX 
    qml.RX(theta, wires=0) 
    return qml.expval(qml.PauliZ(wires=0)) 
# 3. Generowanie Danych 
# # Tworzymy wektor kƒÖt√≥w theta od 0 do 2*pi (pe≈Çny obr√≥t) 
theta_vals = np.linspace(0, 2 * np.pi, 100) 
# Obliczamy warto≈õƒá oczekiwanƒÖ dla ka≈ºdego kƒÖta 
expval_vals = [continuous_expval(theta) for theta in theta_vals] 
# 4. Wizualizacja Wynik√≥w 
plt.figure(figsize=(8, 5)) 
plt.plot(theta_vals, expval_vals, label=r'$\langle \mathbf{Z} \rangle = \cos(\theta)$', color='darkred', linewidth=3)
